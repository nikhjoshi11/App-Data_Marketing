def histogram(column) :
    En_customers = go.Histogram(x  = enrolled[column],
                          histnorm= "percent",
                          name = "Enrolled Customers",
                          marker = dict(line = dict(width = .5,
                                                    color = "black"
                                                    )
                                        ),
                               )
    
    NonEn_customers = go.Histogram(x  = not_enrolled[column],
                          histnorm = "percent",
                          name = "Not Enrolled customers",
                          marker = dict(line = dict(width = .5,
                                              color = "black"
                                             )
                                 ),
                        )
    
    data = [En_customers,NonEn_customers]
    layout = go.Layout(dict(title =column + " - Customer Churn ",
                           
                            xaxis = dict(title = column,
                                         ticklen=5,
                                         
                                            ),
                            yaxis = dict(
                                             title = "percent",
                                             ticklen=5,
                                             
                                            ),
                           )
                      )
    fig  = go.Figure(data=data,layout=layout)
    
    py.iplot(fig)
